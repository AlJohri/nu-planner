package models;

import com.avaje.ebean.config.ServerConfig; import com.avaje.ebean.event.ServerConfigStartup; import com.avaje.ebean.event.BeanPersistListener; import com.avaje.ebean.config.dbplatform.*;

public class MyServerConfigStartup implements ServerConfigStartup { @Override public void onStart(ServerConfig config) {

final ServerConfig config = new ServerConfig();

final PostgresPlatform postgresPlatform = new PostgresPlatform(); postgresPlatform.getDbIdentity().setIdType(IdType.IDENTITY); postgresPlatform.getDbIdentity().setSupportsIdentity(true); config.setDatabasePlatform(postgresPlatform);

serverConfig.add(new BeanPersistListener() { });

} }
